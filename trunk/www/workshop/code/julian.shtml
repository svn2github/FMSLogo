TO SETPEN :BRIGHTNESS
  SETPENCOLOR ( LIST
     255                ; red
     255 - :BRIGHTNESS  ; green
     :BRIGHTNESS        ; blue
  )
END

TO SQUARE
  REPEAT 4 [ FORWARD 50 RIGHT 90 ]
END

TO SQUAREFLOWER
  REPEAT 64 [
    SETPEN REPCOUNT * 4 - 1
    RIGHT 360 / 64
    SQUARE
  ]
END

SQUAREFLOWER
